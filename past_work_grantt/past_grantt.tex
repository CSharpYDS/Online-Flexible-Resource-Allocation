%-----Text Box
%-----#1 height, #2 width, #3 anchor for the label, #4 name of the node, #5
%-----coordinate, #6 label
\def\tbox[#1,#2,#3,#4,#5]#6{%
  \node[draw, drop shadow={opacity=.35}, minimum height=#1, minimum width=#2, %
  inner color=blue!5, outer color=blue!5, color=blue!40!black] (#4) at #5 {}; %
  \node[anchor=#3,inner sep=2pt] at (#4.#3) {#6};%
}
%-----Rotated Text Box
%-----#1 height, #2 width, #3 anchor for the label, #4 name of the node, #5
%-----coordinate, #6 label
\def\rtbox[#1,#2,#3,#4,#5]#6{%
  \node[draw, drop shadow={opacity=.35}, minimum height=#1, minimum width=#2, %
  inner color=blue!5, outer color=blue!5, color=blue!40!black, rotate=90] (#4) at #5 {}; %
  \node[anchor=#3,inner sep=2pt, rotate=90] at (#4.#3) {#6};%
}

%-----Edge
%-----#1 from node, #2 to node, #3 specification of a node (label), #4
%-----dashed, or other parameters for draw
\def\isedge[#1,#2,#3,#4];{ 
  \draw[->,color=black!20!black,#4,fill=white] (#1) -- #3
  (#2);  
}
%-----Grouping box
%-----#1 height, #2 width, #3 name of the node, #4
%-----coordinate, #5 label
\def\gbox[#1,#2,#3,#4,#5]#6{
        \draw[dashed] node[draw,color=gray!50,minimum
        height=#1,minimum width=#2] (#4) at #5 {}; 
        \node[anchor=#3,inner sep=2pt] at (#4.#3)  {#6};
}
